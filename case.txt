CASE [Select Product]
WHEN 'BANK SERVICES' THEN [Product (group)] = 'BANK SERVICES'
WHEN 'CHECKING/SERVICES' THEN [Product (group)] = 'CHECKING/SERVICES'
WHEN 'CREDIT CARD' THEN [Product (group)] = 'CREDIT CARD'
WHEN 'MORTGAGE' THEN [Product (group)] = 'MORTGAGE'
WHEN 'OTHERS' THEN [Product (group)] = 'OTHERS'
WHEN 'ALL' THEN [Product (group)] = ("BANK SERVICES","CHECKING/SERVICES","CREDIT CARD", "MORTGAGE","OTHERS")
END
